const sizeLabels  = {1:'Мини <30см',2:'Средний 30–80',3:'Крупный 80–150',4:'Гигант 150см+'};

const careLabels  = {1:'Очень просто',2:'Просто',3:'Средне',4:'Сложно',5:'Оч. сложно'};

const growLabels  = {1:'Оч. медленный',2:'Медленный',3:'Средний',4:'Быстрый',5:'Оч. быстрый'};


function pips(val, col) {

  let cls;

  if (col==='care'){ const v=6-val; cls=v>=5?'c5':v===4?'c4':v===3?'c3':v===2?'c2':'c1'; }

  else { cls=val===5?'c5':val===4?'c4':val===3?'c3':val===2?'c2':'c1'; }

  let h='<div class="pips">';

  for(let i=1;i<=5;i++) h+=`<span class="pip ${i<=val?cls:''}"></span>`;

  return h+'</div>';

}


function growBar(v) {

  const cls = v>=4?'gon-fast':v===3?'gon-mid':'gon-slow';

  let b='<span class="gbar">';

  for(let i=1;i<=5;i++) b+=`<span class="gb ${i<=v?cls:''}"></span>`;

  return b+`</span><span class="glabel">${growLabels[v]}</span>`;

}


function pbCls(v){ return v<=6?'pb-fast':v<=12?'pb-mid':'pb-slow'; }

function tCls(v){ return v>=26?'t-high':v>=20?'t-mid':'t-low'; }

function multColor(v){ return v>=15?'var(--accent2)':v>=8?'var(--accent)':v>=5?'#b0a060':'var(--muted)'; }

let sortCol='total', sortDir='desc', searchText='', filterType='all';


function getSorted(){

  let arr=[...data];

  if(filterType==='variegated') arr=arr.filter(d=>/variegat|albo|aurea|mint|stardust|halfmoon|splash|multicolor|pink.*var|lava|snowflake/i.test(d.name));

  else if(filterType==='species') arr=arr.filter(d=>!/variegat|albo|aurea|mint|stardust|halfmoon|splash|multicolor|lava|snowflake/i.test(d.name));

  else if(filterType==='highprice') arr=arr.filter(d=>d.price>400);

  else if(filterType==='budget') arr=arr.filter(d=>d.price<=100);

  else if(filterType==='fastpayback') arr=arr.filter(d=>d.payback<=6);

  if(searchText){ const q=searchText.toLowerCase(); arr=arr.filter(d=>d.name.toLowerCase().includes(q)); }

  arr.sort((a,b)=>{

    let va=a[sortCol], vb=b[sortCol];

    if(typeof va==='string'){ va=va.toLowerCase(); vb=vb.toLowerCase(); return sortDir==='asc'?va.localeCompare(vb):vb.localeCompare(va); }

    if(sortCol==='care'||sortCol==='payback') return sortDir==='desc'?va-vb:vb-va;

    return sortDir==='desc'?vb-va:va-vb;

  });

  if(filterType==='top20'){

    const t20=new Set([...data].sort((a,b)=>b.total-a.total).slice(0,20).map(d=>d.num+'-'+d.name));

    arr=arr.filter(d=>t20.has(d.num+'-'+d.name));

  }

  return arr;

}
